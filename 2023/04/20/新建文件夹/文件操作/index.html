    • int main()
{
    FILE* pr=NULL;
    char arr[10];
    //文件的创建
    pr=fopen("chg.tet","r+");
    //文件的判断
    if(pr==NULL)
    {
        perror("fopen:");
        return 1;
    }
    //第一次文件操作
    fputs("chgchg",pr);
    rewind(pr);
    fgets(arr,7,pr);
    printf("%s\n",arr);
    //第二次文件操作
    rewind(pr);
    fputs("tsytsy",pr);
    rewind(pr);
    fgets(arr,7,pr);
    printf("%s\n",arr);
    //文件的关闭
    fclose(pr);

    return 0;
}

二进制写入：

  1. fwrite:将buffer的东西存储进流里面

typedef struct student
{
    char number[20];
    char name[10];
}stu;

int main()
{
    //文件的创建
    FILE* pr = NULL;
    pr = fopen("chg.txt", "r+");
    stu stu1;
    scanf("%s%s", stu1.name, stu1.number);
    //文件的2进制读写
    fwrite(&stu1, sizeof(stu), 1, pr);

    //文件的关闭
    fclose(pr);
    pr = NULL;
    return 0;
}
